// =============================== // INTEL OPS EXTRA MODULES // =============================== // كلمات تحليل التهديد const threatWords = [ "war","attack","missile","strike","bomb","drone", "raid","military","troops","explosion","conflict" ]; // تحليل المشاعر function analyzeSentiment(text){ text = text.toLowerCase(); let score = 0; threatWords.forEach(w=>{ if(text.includes(w)) score+=10; }); if(score > 40) return {level:"CRITICAL",color:"#ff2d2d"}; if(score > 25) return {level:"HIGH",color:"#ff8c00"}; if(score > 10) return {level:"MEDIUM",color:"#ffe600"}; return {level:"LOW",color:"#00ff88"}; } // تحديد الموقع من الخبر function detectLocation(text){ text = text.toLowerCase(); if(text.includes("gaza")) return [31.5,34.8,"غزة"]; if(text.includes("iran")) return [32,53,"إيران"]; if(text.includes("syria")) return [33.5,36.3,"سوريا"]; if(text.includes("iraq")) return [33.3,44.4,"العراق"]; if(text.includes("israel")) return [31,35,"إسرائيل"]; if(text.includes("lebanon")) return [33.8,35.8,"لبنان"]; if(text.includes("yemen")) return [15.3,44.2,"اليمن"]; return null; } // توقع التصعيد function predictEscalation(score){ if(score > 40) return "⚠ تصعيد عسكري محتمل خلال 24 ساعة"; if(score > 20) return "توتر متزايد"; return "الوضع مستقر نسبياً"; } // تحليل الخبر function processNews(title){ const sentiment = analyzeSentiment(title); const location = detectLocation(title); const prediction = predictEscalation(sentiment.level); if(location){ addMarker( location[0], location[1], sentiment.color, location[2], 40, sentiment.level ); highlightCountry( location[0], location[1], sentiment.color ); } return { sentiment, prediction }; } // اختبار النظام function demoIntel(){ const demoNews = [ "Missile strike reported near Gaza border", "Military tensions rising between Iran and Israel", "Explosion heard in Damascus Syria", "Drone attack reported in Iraq base" ]; demoNews.forEach(n=>{ const result = processNews(n); console.log( "خبر:", n, "تحليل:", result ); }); } // تشغيل تلقائي setTimeout(demoIntel,4000);
